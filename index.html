示例代码

    print(i)  # 输出: 0, 1, 2, 3, 4

    print(count)
    count += 1

    if i == 5:
        break  # 终止循环
    if i % 2 == 0:
        continue  # 跳过当前迭代
    print(i)

关卡 9: 异常处理

学习如何处理程序运行中的异常

学习目标

  • 了解异常的基本概念
  • 学习try-except语句的使用
  • 掌握异常处理的最佳实践

示例代码

# 基本异常处理
try:
    num = int(input("请输入一个数字: "))
    result = 10 / num
    print("结果:", result)
except ValueError:
    print("错误: 请输入有效的数字")
except ZeroDivisionError:
    print("错误: 除数不能为零")
except Exception as e:
    print("发生了错误:", str(e))
finally:
    print("程序执行完毕")

示例项目：猜数字游戏

import random

def guess_number():
    """猜数字游戏"""
    number = random.randint(1, 100)
    attempts = 0
    
    print("欢迎来到猜数字游戏！")
    print("我已经想好了一个1到100之间的数字，你有10次机会猜它。")
    
    while attempts < 10:
        try:
            guess = int(input("请输入你的猜测: "))
            attempts += 1
            
            if guess < number:
                print("太小了！")
            elif guess > number:
                print("太大了！")
            else:
                print(f"恭喜你，猜对了！你用了{attempts}次机会。")
                return
        except ValueError:
            print("请输入有效的数字！")
    
    print(f"很遗憾，你没有在10次机会内猜对。正确答案是{number}。")

# 运行游戏
guess_number()
